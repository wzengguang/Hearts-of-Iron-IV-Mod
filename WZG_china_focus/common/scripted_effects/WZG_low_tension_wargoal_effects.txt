WZG_create_annex_wargoal_against_prev = {
	if = {
		limit = { has_country_flag = PRC_low_tension_annex_wargoals }
		create_wargoal = {
			type = PRC_annex_everything_low_tension
			target = PREV
		}
		PREV = { set_country_flag = PRC_low_tension_annex_wargoal_target }
	}
	else = {
		create_wargoal = {
			type = annex_everything
			target = PREV
		}
	}
}
